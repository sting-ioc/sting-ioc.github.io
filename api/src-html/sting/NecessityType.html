001package sting;
002
003/**
004 * Enum describing whether a service is required or optional.
005 */
006public enum NecessityType
007{
008  /**
009   * The service is required.
010   *
011   * <p>If the {@link Service} annotation is describing an input into the service and the injector can
012   * not supply it then compilation of the injector should fail. If the {@link Service} annotation is
013   * describing a service provided then the service is always provided.</p>
014   *
015   * <p>The associated parameter or method may be annotated with {@link javax.annotation.Nonnull}
016   * and must not be annotated with {@link javax.annotation.Nullable}.</p>
017   */
018  REQUIRED,
019
020  /**
021   * The service is optional.
022   *
023   * <p>If the {@link Service} annotation is describing a service provided then the {@link Provides}
024   * method may return a null. If the {@link Service} annotation is describing an input into a service
025   * then the input may be null. The value must not be a nullable value (i.e. not a primitive).</p>
026   *
027   * <p>The associated parameter or method should be annotated with {@link javax.annotation.Nullable}
028   * and must not be annotated with {@link javax.annotation.Nonnull}.</p>
029   */
030  OPTIONAL,
031
032  /**
033   * The service is required or optional based on heuristics.
034   *
035   * <p>If the parameter or method is annotated with {@link javax.annotation.Nullable} and the type is
036    * not primitive, not a {@link java.util.function.Supplier} nor a {@link java.util.Collection} then the
037    * service is considered {@link #OPTIONAL} otherwise it is considered {@link #REQUIRED}.</p>
038   */
039  AUTODETECT
040}
