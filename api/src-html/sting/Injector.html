001package sting;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Retention;
006import java.lang.annotation.RetentionPolicy;
007import java.lang.annotation.Target;
008
009/**
010 * Annotates an interface or abstract class for which a fully-formed, dependency-injected
011 * implementation is to be generated from the {@link #includes() included} object graph fragments. The generated class will
012 * have the name of the type annotated with {@code @Injector} prepended with {@code Sting_}. For
013 * example, {@code @Injector interface MyInjector {...}} will produce an implementation named
014 * {@code Sting_MyInjector}.
015 *
016 * <a name="component-methods"></a>
017 * <h2>Component methods</h2>
018 *
019 * <p>Every type annotated with {@code @Injector} must contain at least one abstract component
020 * method. Every abstract method must return a type provided by the object graph and must not
021 * have any parameters. The dependency should be provisioned by a {@link Injectable @Injectable}
022 * annotated type or a {@link Provides @Provides} annotated method.
023 *
024 * <a name="instantiation"></a>
025 * <h2>Instantiation</h2>
026 *
027 * <p>If a nested interface named Factory exists in the component, an implementation of the Factory interface
028 * will be generated and an instance will be returned via a static method named {@code factory()}.</p>
029 *
030 * <p>Example of using a factory:</p>
031 *
032 * <pre><code>
033 * {@literal @}Injector(includes = {BackendFragment.class, FrontendFragment.class})
034 * interface MyInjector {
035 *   MyWidget myWidget();
036 *
037 *   interface Factory {
038 *     MyInjector create(MyService myService);
039 *   }
040 * }
041 *
042 * public class Main {
043 *   public static void main(String[] args) {
044 *     MyService myService = ...;
045 *     MyInjector injector = Sting_MyInjector.factory().create(myService);
046 *   }
047 * }</code></pre>
048 *
049 *
050 * <p>If a nested interface named Factory does not exist then it is assumed that the component
051 * is self-contained and the generated component will have a factory method {@code create()}.</p>
052 *
053 * <p>Example of using create:</p>
054 *
055 * <pre><code>
056 * {@literal @}Injector(includes = {BackendFragment.class, FrontendFragment.class})
057 * interface MyInjector {
058 *   MyWidget myWidget();
059 * }
060 *
061 * public class Main {
062 *   public static void main(String[] args) {
063 *     MyInjector injector = StingMyInjector.create();
064 *   }
065 * }</code></pre>
066 */
067@Documented
068@Retention( RetentionPolicy.RUNTIME )
069@Target( ElementType.TYPE )
070public @interface Injector
071{
072  /**
073   * A list of types that contribute to the object graph.
074   * These types can be {@code @Fragment}-annotated interfaces or {@link Injectable @Injectable}-annotated classes.
075   * The de-duplicated contributions of the {@code @Fragment}-annotated interfaces in the
076   * {@code includes}, and of their inclusions recursively, are all contributed
077   * to the object graph.
078   *
079   * <p>If the annotation processor detects a dependency that is required but not explicitly included in the
080   * includes list then it will attempt to automatically add the type to the graph if it is annotated with
081   * {@link Injectable @Injectable}. The current implementation include types
082   * if they were compiled in the same invocation of the java compiler. In the future the annotation processor
083   * will load the descriptors from the filesystem.</p>
084   *
085   * @return a list of types that contribute to the injectors object graph.
086   */
087  Class<?>[] includes() default {};
088
089  /**
090   * A list of services that must be passed into the injector.
091   * The annotation processor will generate a constructor with one parameter for every input. Each input
092   * value MUST specify the type parameter otherwise the annotation processor is unable to determine the
093   * type of the binding.
094   *
095   * @return a list of services that must be passed into the injector.
096   */
097  Service[] inputs() default {};
098}
