001package sting;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Retention;
006import java.lang.annotation.RetentionPolicy;
007import java.lang.annotation.Target;
008import java.util.function.Supplier;
009import javax.annotation.Nonnull;
010import javax.annotation.Nullable;
011
012/**
013 * Identifies an injectable type.
014 * Types annotated with this are expected to be created by an injector.
015 * The type must be concrete and must have a single package-access constructor. The constructor can accept zero or
016 * more dependencies as arguments.
017 *
018 * <b>Optional Dependencies</b>
019 *
020 * <p>A non-primitive parameter can be marked as optional by annotating it with the
021 * {@linkplain Nullable @Nullable} annotation. If the injector can not find a value to satisfy the dependency
022 * then the injector may pass null for the parameter.</p>
023 *
024 * <b>Qualified Dependencies</b>
025 *
026 * <p>A parameter may can also be annotated with the {@link Dependency @Dependency} annotation with the qualifier
027 * parameter specified. To satisfy the dependency the provided value must be declared with the same qualifier.</p>
028 *
029 * <h3>Circular Dependencies</h3>
030 *
031 * <p>Circular dependencies are disallowed by the injector and are rejected during the compilation phase.
032 * The developer can break the circular dependency by injecting {@link Supplier Supplier&lt;OtherType>}
033 * instead of {@code OtherType} and then calling {@link Supplier#get() get()} on the supplier when access
034 * to the dependency is needed.</p>
035 *
036 * @see Dependency @Dependency
037 * @see Supplier
038 */
039@Target( ElementType.TYPE )
040@Retention( RetentionPolicy.RUNTIME )
041@Documented
042public @interface Injectable
043{
044  /**
045   * A unique identifier representing this binding.
046   * If not identified then this will be derived as "[Qualified Classname]".
047   * This is primarily used so that inherited {@link Injector} annotated types can override specific bindings.
048   *
049   * @return a unique identifier representing this binding..
050   */
051  @Nonnull
052  String id() default "";
053
054  /**
055   * A flag indicating whether the instance should be instantiated when the Injector is created.
056   *
057   * @return a flag indicating whether the instance should be instantiated when the Injector is created.
058   */
059  boolean eager() default false;
060
061  /**
062   * An opaque string that qualifies this type.
063   * This can be any arbitrary string and is used to restrict the dependencies that this type can satisfy.
064   *
065   * @return an opaque qualifier string.
066   */
067  @Nonnull
068  String qualifier() default "";
069
070  /**
071   * The types of dependency that this binding can satisfy.
072   * By default the binding will match the type that is annotated.
073   * If specified, the binding will only match the types specified and
074   * every type must be assignable from the type that is annotated.
075   *
076   * @return the types of dependency that this binding can satisfy.
077   */
078  Class<?>[] types() default void.class;
079}
