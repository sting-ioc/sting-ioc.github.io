001package sting;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Retention;
006import java.lang.annotation.RetentionPolicy;
007import java.lang.annotation.Target;
008
009/**
010 * Identify an interface that can contribute to a component graph.
011 * The interface is expected to declare 1 or more default methods and/or
012 * include 1 or more types in the {@link #includes() includes} parameter.
013 */
014@Documented
015@Retention( RetentionPolicy.RUNTIME )
016@Target( ElementType.TYPE )
017public @interface Fragment
018{
019  /**
020   * A list of types that can contribute to the component graph.
021   * The types can be {@link Injectable}-annotated classes or {@link Fragment}-annotated interfaces.
022   * The {@link Fragment}-annotated interfaces contributions are added recursively and contributions are
023   * de-duplicated before they are resolved.
024   *
025   * @return a list of types that contribute to the fragments component graph.
026   */
027  Class<?>[] includes() default {};
028}
