001package sting;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Retention;
006import java.lang.annotation.RetentionPolicy;
007import java.lang.annotation.Target;
008
009/**
010 * Annotates an interface that contributes to the object graph.
011 * The interface is expected to declare 1 or more {@link Provides provider} methods and/or
012 * include 1 or more types in the {@link #includes() includes} parameter.
013 * As this annotation is can only be applied to an interface, the {@link Provides provider} methods must
014 * be <code>default</code> methods. Tne methods may be annotated with {@link Provides @Provides}
015 * but even if they are not they are expected to comply with the requirements of
016 * {@link Provides @Provides} annotated methods.
017 */
018@Documented
019@Retention( RetentionPolicy.RUNTIME )
020@Target( ElementType.TYPE )
021public @interface Fragment
022{
023  /**
024   * A list of types that contribute to the object graph.
025   * These types can be {@code @Fragment}-annotated interfaces or {@link Injectable @Injectable}-annotated classes.
026   * The de-duplicated contributions of the {@code @Fragment}-annotated interfaces in the
027   * {@code includes}, and of their inclusions recursively, are all contributed
028   * to the object graph.
029   *
030   * @return a list of types that contribute to the fragments object graph.
031   */
032  Class<?>[] includes() default {};
033}
