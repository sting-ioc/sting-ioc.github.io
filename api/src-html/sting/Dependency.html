001package sting;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Retention;
006import java.lang.annotation.RetentionPolicy;
007import java.lang.annotation.Target;
008import javax.annotation.Nonnull;
009
010/**
011 * Annotation used to describe the dependency.
012 * The annotation parameters restrict the bindings that can satisfy the dependency.
013 *
014 * <p>This annotation can appear on constructor parameters in a type annotated with {@link Injectable}
015 * or it can appear on methods on types annotated by {@link Injector}.
016 * If a dependency is annotated with {@code @Dependency} then the dependency is only satisfied by an
017 * {@link Injectable} annotated type or a {@link Provides} annotated method with matching qualifier
018 * attribute and a matching dependency type.
019 */
020@Target( { ElementType.PARAMETER, ElementType.METHOD } )
021@Retention( RetentionPolicy.RUNTIME )
022@Documented
023public @interface Dependency
024{
025  /**
026   * An opaque string that qualifies the dependency.
027   *
028   * @return an opaque qualifier string.
029   */
030  @Nonnull
031  String qualifier() default "";
032
033  /**
034   * The type of the dependency required.
035   *
036   * <p>If the {@code @Dependency} annotation is attached to a constructor or method parameter then the
037   * default value of the annotation parameter is the the type of the constructor or method parameter.
038   * If this annotation parameter is explicitly specified then the value MUST be assignable to the type of
039   * the constructor or method parameter.</p>
040   *
041   * <p>If the {@code @Dependency} annotation is attached to a method then the default value of the annotation
042   * parameter is the the return type of the method. If the annotation parameter is explicitly specified then the
043   * value MUST be assignable to the return type of the method.</p>
044   *
045   * @return the type of the dependency required.
046   */
047  Class<?> type() default void.class;
048
049  /**
050   * A parameter indicating whether the service is required or optional.
051   *
052   * @return a parameter indicating whether the service is required or optional.
053   */
054  @Nonnull
055  NecessityType necessity() default NecessityType.AUTODETECT;
056}
