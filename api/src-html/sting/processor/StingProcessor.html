152  @Override
153  public boolean process( @Nonnull final Set<? extends TypeElement> annotations, @Nonnull final RoundEnvironment env )
154  {
155    processTypeElements( annotations,
156                         env,
157                         Constants.INJECTABLE_CLASSNAME,
158                         _deferredInjectableTypes,
159                         this::processInjectable );
160
161    processTypeElements( annotations,
162                         env,
163                         Constants.FRAGMENT_CLASSNAME,
164                         _deferredFragmentTypes,
165                         this::processFragment );
166
167    annotations.stream()
168      .filter( a -> a.getQualifiedName().toString().equals( Constants.NAMED_CLASSNAME ) )
169      .findAny()
170      .ifPresent( a -> verifyNamedElements( env, env.getElementsAnnotatedWith( a ) ) );
171
172    annotations.stream()
173      .filter( a -> a.getQualifiedName().toString().equals( Constants.TYPED_CLASSNAME ) )
174      .findAny()
175      .ifPresent( a -> verifyTypedElements( env, env.getElementsAnnotatedWith( a ) ) );
176
177    annotations.stream()
178      .filter( a -> a.getQualifiedName().toString().equals( Constants.EAGER_CLASSNAME ) )
179      .findAny()
180      .ifPresent( a -> verifyEagerElements( env, env.getElementsAnnotatedWith( a ) ) );
181
182    processTypeElements( annotations,
183                         env,
184                         Constants.INJECTOR_CLASSNAME,
185                         _deferredInjectorTypes,
186                         this::processInjector );
187
188    processResolvedInjectables( env );
189    processResolvedFragments( env );
190    processResolvedInjectors( env );
191
192    errorIfProcessingOverAndInvalidTypesDetected( env );
193    errorIfProcessingOverAndUnprocessedInjectorDetected( env );
194    if ( env.processingOver() || env.errorRaised() )
195    {
196      _registry.clear();
197    }
198    return true;
199  }

201  private void errorIfProcessingOverAndUnprocessedInjectorDetected( @Nonnull final RoundEnvironment env )
202  {
203    if ( env.processingOver() && !env.errorRaised() )
204    {
205      final List<InjectorDescriptor> injectors = _registry.getInjectors();
206      if ( !injectors.isEmpty() )
207      {
208        processingEnv
209          .getMessager()
210          .printMessage( Diagnostic.Kind.ERROR,
211                         getClass().getSimpleName() + " failed to process " + injectors.size() + " injectors " +
212                         "as not all of their dependencies could be resolved. The java code resolved but the " +
213                         "descriptors were missing or in the incorrect format. Ensure that the included " +
214                         "typed have been compiled with a compatible version of Sting and that the .sbf " +
215                         "descriptors have been packaged with the .class files." );
216        for ( final InjectorDescriptor injector : injectors )
217        {
218          processingEnv
219            .getMessager()
220            .printMessage( Diagnostic.Kind.ERROR,
221                           "Failed to process the " + injector.getElement().getQualifiedName() + " injector." );
222        }
223      }
224    }
225  }

228  {

280  {

293  private void buildAndEmitObjectGraph( @Nonnull final InjectorDescriptor injector )
294    throws Exception
295  {
296    final ComponentGraph graph = new ComponentGraph( injector );
297    registerIncludesComponents( graph );
298
299    registerInputs( graph );
300
301    buildObjectGraphNodes( graph );
302
303    if ( graph.getNodes().isEmpty() && graph.getRootNode().getDependsOn().isEmpty() )
304    {
305      throw new ProcessorException( MemberChecks.toSimpleName( Constants.INJECTOR_CLASSNAME ) + " target " +
306                                    "produced an empty object graph. This means that there are no eager nodes " +
307                                    "in the includes and there are no dependencies or only unsatisfied optional " +
308                                    "dependencies defined by the injector",
309                                    graph.getInjector().getElement() );
310    }
311
312    propagateEagerFlagUpstream( graph );
313
314    CircularDependencyChecker.verifyNoCircularDependencyLoops( graph );
315
316    emitObjectGraphJsonDescriptor( graph );
317
318    final String packageName = GeneratorUtil.getQualifiedPackageName( graph.getInjector().getElement() );
319    emitTypeSpec( packageName, InjectorGenerator.buildType( processingEnv, graph ) );
320    if ( injector.isInjectable() )
321    {
322      emitTypeSpec( packageName, InjectorProviderGenerator.buildType( processingEnv, graph ) );
323    }
324    emitDotReport( graph );
325  }

347  private void propagateEagerFlagUpstream( @Nonnull final ComponentGraph graph )
348  {
349    // Propagate Eager flag to all dependencies of eager nodes breaking the propagation at Supplier nodes
350    // They may not be configured as eager but they are effectively eager given that they will be created
351    // at startup, they may as well be marked as eager objects as that results in smaller code-size.
352    graph.getNodes().stream().filter( n -> n.getBinding().isEager() ).forEach( Node::markNodeAndUpstreamAsEager );
353  }

360  private void registerIncludes( @Nonnull final ComponentGraph graph,
361                                 @Nonnull final Collection<IncludeDescriptor> includes )
362  {
363    for ( final IncludeDescriptor include : includes )
364    {
365      final String classname = include.getActualTypeName();
366      final TypeElement element = processingEnv.getElementUtils().getTypeElement( classname );
367      assert null != element;
368      if ( AnnotationsUtil.hasAnnotationOfType( element, Constants.FRAGMENT_CLASSNAME ) )
369      {
370        final FragmentDescriptor fragment = _registry.getFragmentByClassName( element.getQualifiedName().toString() );
371        registerIncludes( graph, fragment.getIncludes() );
372        graph.registerFragment( fragment );
373      }
374      else
375      {
376        assert AnnotationsUtil.hasAnnotationOfType( element, Constants.INJECTABLE_CLASSNAME );
377        graph.registerInjectable( _registry.getInjectableByClassName( element.getQualifiedName().toString() ) );
378      }
379    }
380  }

382  private void buildObjectGraphNodes( @Nonnull final ComponentGraph graph )
383  {
384    final Set<Node> completed = new HashSet<>();
385    final Stack<WorkEntry> workList = new Stack<>();
386    // At this stage the "rootNode" contains dependencies for all the service methods declared on the injector
387    // and all the eager services declared in includes have already been added to the nodes list.
388    //
389    // We start at the rootNode and expand all of the dependencies. And then we take any of the eager dependencies
390    // that have yet to be added to be processed and add them to to worklist and expand all dependencies until there
391    // are no eager nodes left to process
392    final List<Node> eagerNodes = new ArrayList<>( graph.getRawNodeCollection() );
393    final Node rootNode = graph.getRootNode();
394    rootNode.setDepth( 0 );
395    addDependsOnToWorkList( workList, rootNode, null );
396    processWorkList( graph, completed, workList );
397    for ( final Node node : eagerNodes )
398    {
399      if ( node.isDepthNotSet() )
400      {
401        node.setDepth( 0 );
402        addDependsOnToWorkList( workList, node, null );
403        processWorkList( graph, completed, workList );
404      }
405    }
406    graph.complete();
407  }

409  private void processWorkList( @Nonnull final ComponentGraph graph,
410                                @Nonnull final Set<Node> completed,
411                                @Nonnull final Stack<WorkEntry> workList )
412  {
413    final InjectorDescriptor injector = graph.getInjector();
414    while ( !workList.isEmpty() )
415    {
416      final WorkEntry workEntry = workList.pop();
417      final Edge edge = workEntry.getEntry().getEdge();
418      assert null != edge;
419      final ServiceDescriptor service = edge.getService();
420      final Coordinate coordinate = service.getService().getCoordinate();
421      final List<Binding> bindings = new ArrayList<>( graph.findAllBindingsByCoordinate( coordinate ) );
422
423      if ( bindings.isEmpty() )
424      {
425        final String classname = coordinate.getType().toString();
426        final InjectableDescriptor injectable = _registry.findInjectableByClassName( classname );
427        if ( null != injectable &&
428             injectable.getBinding()
429               .getPublishedServices()
430               .stream()
431               .anyMatch( s -> coordinate.equals( s.getCoordinate() ) ) )
432        {
433          bindings.add( injectable.getBinding() );
434        }
435        if ( bindings.isEmpty() )
436        {
437          final TypeElement typeElement = processingEnv.getElementUtils().getTypeElement( classname );
438          final byte[] data = tryLoadDescriptorData( typeElement );
439          if ( null != data )
440          {
441            final Node node = edge.getNode();
442            final Object owner = node.hasNoBinding() ? null : node.getBinding().getOwner();
443            final TypeElement ownerElement =
444              owner instanceof FragmentDescriptor ? ( (FragmentDescriptor) owner ).getElement() :
445              owner instanceof InjectableDescriptor ? ( (InjectableDescriptor) owner ).getElement() :
446              injector.getElement();
447
448            final Object descriptor = loadDescriptor( ownerElement, classname, data );
449            if ( descriptor instanceof InjectableDescriptor )
450            {
451              final InjectableDescriptor injectableDescriptor = (InjectableDescriptor) descriptor;
452              if ( injectableDescriptor.getBinding()
453                .getPublishedServices()
454                .stream()
455                .anyMatch( s -> coordinate.equals( s.getCoordinate() ) ) )
456              {
457                _registry.registerInjectable( injectableDescriptor );
458                bindings.add( injectableDescriptor.getBinding() );
459              }
460            }
461          }
462        }
463      }
464
465      final List<Binding> nullableProviders = bindings.stream()
466        .filter( b -> b.getPublishedServices().stream().anyMatch( ServiceSpec::isOptional ) )
467        .collect( Collectors.toList() );
468      if ( !service.getService().isOptional() && !nullableProviders.isEmpty() )
469      {
470        final String message =
471          MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
472                                "contain an optional provider method or optional injector input and " +
473                                "a non-optional service request for the coordinate " + coordinate + "\n" +
474                                "Dependency Path:\n" + workEntry.describePathFromRoot() + "\n" +
475                                "Binding" + ( nullableProviders.size() > 1 ? "s" : "" ) + ":\n" +
476                                bindingsToString( nullableProviders ) );
477        throw new ProcessorException( message, service.getElement() );
478      }
479      if ( bindings.isEmpty() )
480      {
481        if ( service.getService().isOptional() || service.getKind().isCollection() )
482        {
483          edge.setSatisfiedBy( Collections.emptyList() );
484        }
485        else
486        {
487          final String message =
488            MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
489                                  "contain a non-optional dependency " + coordinate +
490                                  " that can not be satisfied.\n" +
491                                  "Dependency Path:\n" + workEntry.describePathFromRoot() );
492          throw new ProcessorException( message, service.getElement() );
493        }
494      }
495      else
496      {
497        final ServiceDescriptor.Kind kind = service.getKind();
498        if ( 1 == bindings.size() || kind.isCollection() )
499        {
500          final List<Node> nodes = new ArrayList<>();
501          for ( final Binding binding : bindings )
502          {
503            final Node node = graph.findOrCreateNode( binding );
504            nodes.add( node );
505            if ( !completed.contains( node ) )
506            {
507              completed.add( node );
508              addDependsOnToWorkList( workList, node, workEntry );
509            }
510          }
511          edge.setSatisfiedBy( nodes );
512        }
513        else
514        {
515          //noinspection ConstantConditions
516          assert bindings.size() > 1 && !kind.isCollection();
517          final String message =
518            MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
519                                  "contain a non-collection dependency " + coordinate +
520                                  " that can be satisfied by multiple nodes.\n" +
521                                  "Dependency Path:\n" + workEntry.describePathFromRoot() + "\n" +
522                                  "Candidate Nodes:\n" + bindingsToString( bindings ) );
523          throw new ProcessorException( message, service.getElement() );
524        }
525      }
526    }
527  }

532    return bindings
533      .stream()
534      .map( b -> "  " + b.getTypeLabel() + "    " + b.describe() )
535      .collect( Collectors.joining( "\n" ) );

538  private void addDependsOnToWorkList( @Nonnull final Stack<WorkEntry> workList,
539                                       @Nonnull final Node node,
540                                       @Nullable final WorkEntry parent )
541  {
542    for ( final Edge e : node.getDependsOn() )
543    {
544      final Stack<PathEntry> stack = new Stack<>();
545      if ( null != parent )
546      {
547        stack.addAll( parent.getStack() );
548      }
549      final PathEntry entry = new PathEntry( node, e );
550      stack.add( entry );
551      workList.add( new WorkEntry( entry, stack ) );
552    }
553  }

588  private boolean isResolved( @Nonnull final RoundEnvironment env,
589                              @Nonnull final Object descriptor,
590                              @Nonnull final TypeElement originator,
591                              @Nonnull final Collection<IncludeDescriptor> includes )
592  {
593    boolean resolved = true;
594    // By the time we get here we can guarantee that the java types are correctly resolved
595    // so we only have to check that the descriptors are present and valid in this method
596    // Except for providers. Providers may be loaded a lot later
597    for ( final IncludeDescriptor include : includes )
598    {
599      final String classname = include.getActualTypeName();
600      final TypeElement element = processingEnv.getElementUtils().getTypeElement( classname );
601      if ( null == element )
602      {
603        assert include.isProvider();
604        if ( env.processingOver() )
605        {
606          AnnotationMirror annotation =
607            AnnotationsUtil.findAnnotationByType( originator, Constants.INJECTOR_CLASSNAME );
608
609          final String annotationClassname =
610            null != annotation ? Constants.INJECTOR_CLASSNAME : Constants.FRAGMENT_CLASSNAME;
611          if ( null == annotation )
612          {
613            annotation = AnnotationsUtil.getAnnotationByType( originator, Constants.FRAGMENT_CLASSNAME );
614          }
615
616          if ( descriptor instanceof FragmentDescriptor )
617          {
618            ( (FragmentDescriptor) descriptor ).markAsContainsError();
619          }
620          else
621          {
622            ( (InjectorDescriptor) descriptor ).markAsContainsError();
623          }
624
625          final String message =
626            MemberChecks.toSimpleName( annotationClassname ) + " target has an " +
627            "parameter named 'includes' containing the value " + include.getIncludedType() +
628            " and that type is annotated by the @StingProvider annotation. The provider annotation expects a " +
629            "provider class named " + include.getActualTypeName() + " but no such class exists. The " +
630            "type need to be removed from the includes or the provider class needs to be present.";
631          reportError( env, message, originator, annotation, null );
632        }
633        return false;
634      }
635      final boolean isInjectable = AnnotationsUtil.hasAnnotationOfType( element, Constants.INJECTABLE_CLASSNAME );
636      final boolean isFragment =
637        !isInjectable && AnnotationsUtil.hasAnnotationOfType( element, Constants.FRAGMENT_CLASSNAME );
638      if ( include.isProvider() && !isInjectable && !isFragment )
639      {
640        AnnotationMirror annotation =
641          AnnotationsUtil.findAnnotationByType( originator, Constants.INJECTOR_CLASSNAME );
642
643        final String annotationClassname =
644          null != annotation ? Constants.INJECTOR_CLASSNAME : Constants.FRAGMENT_CLASSNAME;
645        if ( null == annotation )
646        {
647          annotation = AnnotationsUtil.getAnnotationByType( originator, Constants.FRAGMENT_CLASSNAME );
648        }
649        if ( descriptor instanceof FragmentDescriptor )
650        {
651          ( (FragmentDescriptor) descriptor ).markAsContainsError();
652        }
653        else
654        {
655          ( (InjectorDescriptor) descriptor ).markAsContainsError();
656        }
657
658        final String message =
659          MemberChecks.toSimpleName( annotationClassname ) + " target has an " +
660          "parameter named 'includes' containing the value " + include.getIncludedType() +
661          " and that type is annotated by the @StingProvider annotation. The provider annotation expects a " +
662          "provider class named " + include.getActualTypeName() + " but that class is not annotated with either " +
663          MemberChecks.toSimpleName( Constants.INJECTOR_CLASSNAME ) + " or " +
664          MemberChecks.toSimpleName( Constants.FRAGMENT_CLASSNAME );
665        throw new ProcessorException( message, originator, annotation );
666      }
667      if ( isFragment )
668      {
669        FragmentDescriptor fragment = _registry.findFragmentByClassName( classname );
670        if ( null == fragment )
671        {
672          final byte[] data = tryLoadDescriptorData( element );
673          if ( null == data )
674          {
675            debug( () -> "The fragment " + classname + " is compiled to a .class file but no descriptor is present" );
676            return false;
677          }
678          final Object loadedDescriptor = loadDescriptor( originator, classname, data );
679          if ( loadedDescriptor instanceof FragmentDescriptor )
680          {
681            fragment = (FragmentDescriptor) loadedDescriptor;
682            _registry.registerFragment( fragment );
683          }
684          else
685          {
686            debug( () -> "The fragment " + classname + " is compiled to a .class " +
687                         "file but an invalid descriptor is present" );
688            return false;
689          }
690        }
691        if ( !isFragmentReady( env, fragment ) )
692        {
693          debug( () -> "Fragment include " + classname + " is present but not yet resolved" );
694          resolved = false;
695        }
696      }
697      else
698      {
699        InjectableDescriptor injectable = _registry.findInjectableByClassName( classname );
700        if ( null == injectable )
701        {
702          final byte[] data = tryLoadDescriptorData( element );
703          if ( null == data )
704          {
705            debug( () -> "The injectable " + classname + " is compiled to a .class file but no descriptor is present" );
706            return false;
707          }
708          final Object loadedDescriptor = loadDescriptor( originator, classname, data );
709          if ( loadedDescriptor instanceof InjectableDescriptor )
710          {
711            _registry.registerInjectable( (InjectableDescriptor) loadedDescriptor );
712          }
713          else
714          {
715            debug( () -> "The injectable " + classname + " is compiled to a .class " +
716                         "file but an invalid descriptor is present" );
717            return false;
718          }
719        }
720      }
721    }
722    return resolved;
723  }

725  private void processInjector( @Nonnull final TypeElement element )
726    throws Exception
727  {
728    debug( () -> "Processing Injector: " + element );
729    final ElementKind kind = element.getKind();
730    if ( ElementKind.INTERFACE != kind )
731    {
732      throw new ProcessorException( MemberChecks.must( Constants.INJECTOR_CLASSNAME, "be an interface" ),
733                                    element );
734    }
735    if ( !element.getTypeParameters().isEmpty() )
736    {
737      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME, "have type parameters" ),
738                                    element );
739    }
740    final List<? extends AnnotationMirror> scopedAnnotations = getScopedAnnotations( element );
741    if ( !scopedAnnotations.isEmpty() )
742    {
743      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
744                                                          "be annotated with an annotation that is " +
745                                                          "annotated with the " + Constants.JSR_330_SCOPE_CLASSNAME +
746                                                          " annotation such as " + scopedAnnotations ),
747                                    element );
748    }
749
750    final boolean gwt = isGwtEnabled( element );
751    final boolean injectable =
752      (boolean) AnnotationsUtil.getAnnotationValue( element, Constants.INJECTOR_CLASSNAME, "injectable" ).getValue();
753    final List<IncludeDescriptor> includes = extractIncludes( element, Constants.INJECTOR_CLASSNAME );
754    final List<InputDescriptor> inputs = extractInputs( element );
755
756    final List<ServiceDescriptor> outputs = new ArrayList<>();
757    final List<ExecutableElement> methods =
758      ElementsUtil.getMethods( element, processingEnv.getElementUtils(), processingEnv.getTypeUtils() );
759    for ( final ExecutableElement method : methods )
760    {
761      if ( method.getModifiers().contains( Modifier.ABSTRACT ) )
762      {
763        processInjectorOutputMethod( outputs, method );
764      }
765    }
766    for ( final Element enclosedElement : element.getEnclosedElements() )
767    {
768      if ( ElementKind.INTERFACE == enclosedElement.getKind() &&
769           AnnotationsUtil.hasAnnotationOfType( enclosedElement, Constants.FRAGMENT_CLASSNAME ) )
770      {
771        final DeclaredType type = (DeclaredType) enclosedElement.asType();
772        if ( includes.stream().noneMatch( d -> Objects.equals( d.getIncludedType(), type ) ) )
773        {
774          includes.add( new IncludeDescriptor( type, type.toString() ) );
775        }
776      }
777      else if ( ElementKind.CLASS == enclosedElement.getKind() &&
778                AnnotationsUtil.hasAnnotationOfType( enclosedElement, Constants.INJECTABLE_CLASSNAME ) )
779      {
780        final DeclaredType type = (DeclaredType) enclosedElement.asType();
781        if ( includes.stream().noneMatch( d -> Objects.equals( d.getIncludedType(), type ) ) )
782        {
783          includes.add( new IncludeDescriptor( type, type.toString() ) );
784        }
785      }
786    }
787    final InjectorDescriptor injector = new InjectorDescriptor( element, gwt, injectable, includes, inputs, outputs );
788    _registry.registerInjector( injector );
789    emitInjectorJsonDescriptor( injector );
790  }

811  private void processInjectorOutputMethod( @Nonnull final List<ServiceDescriptor> outputs,
812                                            @Nonnull final ExecutableElement method )
813  {
814    assert method.getModifiers().contains( Modifier.ABSTRACT );
815    if ( TypeKind.VOID == method.getReturnType().getKind() )
816    {
817      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
818                                                          "contain a method that has a void return value" ),
819                                    method );
820    }
821    else if ( !method.getParameters().isEmpty() )
822    {
823      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
824                                                          "contain a method that has parameters" ),
825                                    method );
826    }
827    else if ( !method.getTypeParameters().isEmpty() )
828    {
829      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
830                                                          "contain a method that has any type parameters" ),
831                                    method );
832    }
833    else if ( !method.getThrownTypes().isEmpty() )
834    {
835      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
836                                                          "contain a method that throws any exceptions" ),
837                                    method );
838    }
839    final List<? extends AnnotationMirror> scopedAnnotations = getScopedAnnotations( method );
840    if ( !scopedAnnotations.isEmpty() )
841    {
842      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
843                                                          "contain a method that is annotated with an " +
844                                                          "annotation that is annotated with the " +
845                                                          Constants.JSR_330_SCOPE_CLASSNAME +
846                                                          " annotation such as " + scopedAnnotations ),
847                                    method );
848    }
849    outputs.add( processOutputMethod( method ) );
850  }

852  @Nonnull
853  private ServiceDescriptor processOutputMethod( @Nonnull final ExecutableElement method )
854  {
855    final TypeMirror type = method.getReturnType();
856    if ( TypesUtil.containsArrayType( type ) )
857    {
858      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
859                                                          "contain a method with a return type that contains an array type" ),
860                                    method );
861    }
862    else if ( TypesUtil.containsWildcard( type ) )
863    {
864      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
865                                                          "contain a method with a return type that contains a wildcard type parameter" ),
866                                    method );
867    }
868    else if ( TypesUtil.containsRawType( type ) )
869    {
870      throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
871                                                          "contain a method with a return type that contains a raw type" ),
872                                    method );
873    }
874    else
875    {
876      TypeMirror dependencyType = null;
877      ServiceDescriptor.Kind kind = null;
878      for ( final ServiceDescriptor.Kind candidate : ServiceDescriptor.Kind.values() )
879      {
880        dependencyType = candidate.extractType( type );
881        if ( null != dependencyType )
882        {
883          kind = candidate;
884          break;
885        }
886      }
887      if ( null == kind )
888      {
889        throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
890                                                            "contain a method with a return type that contains an unexpected parameterized type. Only parameterized types known to the framework are supported" ),
891                                      method );
892      }
893      else
894      {
895        final boolean optional = AnnotationsUtil.hasNullableAnnotation( method );
896        if ( optional && ServiceDescriptor.Kind.INSTANCE != kind )
897        {
898          throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
899                                                              "contain a method annotated with " +
900                                                              MemberChecks.toSimpleName( AnnotationsUtil.NULLABLE_CLASSNAME ) +
901                                                              " that is not an instance dependency kind" ),
902                                        method );
903        }
904        final String qualifier = getQualifier( method );
905        if ( AnnotationsUtil.hasAnnotationOfType( method, Constants.JSR_330_NAMED_CLASSNAME ) )
906        {
907          throw new ProcessorException( MemberChecks.mustNot( Constants.INJECTOR_CLASSNAME,
908                                                              "contain a method annotated with the " +
909                                                              Constants.JSR_330_NAMED_CLASSNAME +
910                                                              " annotation. Use the " + Constants.NAMED_CLASSNAME +
911                                                              " annotation instead" ),
912                                        method );
913        }
914
915        final Coordinate coordinate = new Coordinate( qualifier, dependencyType );
916        final ServiceSpec service = new ServiceSpec( coordinate, optional );
917        return new ServiceDescriptor( kind, service, method, -1 );
918      }
919    }
920  }

922  private void verifyNamedElements( @Nonnull final RoundEnvironment env,
923                                    @Nonnull final Set<? extends Element> elements )
924  {
925    for ( final Element element : elements )
926    {
927      if ( ElementKind.PARAMETER == element.getKind() )
928      {
929        final Element executableElement = element.getEnclosingElement();
930        final boolean injectableType =
931          AnnotationsUtil.hasAnnotationOfType( executableElement.getEnclosingElement(),
932                                               Constants.INJECTABLE_CLASSNAME );
933        final boolean isFragmentType =
934          !injectableType &&
935          AnnotationsUtil.hasAnnotationOfType( executableElement.getEnclosingElement(), Constants.FRAGMENT_CLASSNAME );
936        final ElementKind executableKind = executableElement.getKind();
937        final boolean isProvider =
938          !injectableType && !isFragmentType && hasStingProvider( executableElement.getEnclosingElement() );
939        if ( !injectableType && ElementKind.CONSTRUCTOR == executableKind && !isProvider )
940        {
941          reportError( env,
942                       MemberChecks.must( Constants.NAMED_CLASSNAME,
943                                          "only be present on a constructor parameter if the constructor " +
944                                          "is enclosed in a type annotated with " +
945                                          MemberChecks.toSimpleName( Constants.INJECTABLE_CLASSNAME ) +
946                                          " or the type has an associated provider" ),
947                       element );
948        }
949        else if ( !isFragmentType && ElementKind.METHOD == executableKind )
950        {
951          reportError( env,
952                       MemberChecks.must( Constants.NAMED_CLASSNAME,
953                                          "only be present on a method parameter if the method is enclosed in a type annotated with " +
954                                          MemberChecks.toSimpleName( Constants.FRAGMENT_CLASSNAME ) ),
955                       element );
956        }
957        else
958        {
959          assert ( injectableType && ElementKind.CONSTRUCTOR == executableKind ) ||
960                 ( isProvider && ElementKind.CONSTRUCTOR == executableKind ) ||
961                 ( isFragmentType && ElementKind.METHOD == executableKind );
962        }
963      }
964      else if ( ElementKind.CLASS == element.getKind() )
965      {
966        if ( !AnnotationsUtil.hasAnnotationOfType( element, Constants.INJECTABLE_CLASSNAME ) &&
967             !hasStingProvider( element ) )
968        {
969          reportError( env,
970                       MemberChecks.must( Constants.NAMED_CLASSNAME,
971                                          "only be present on a type if the type is annotated with " +
972                                          MemberChecks.toSimpleName( Constants.INJECTABLE_CLASSNAME ) +
973                                          " or the type has an associated provider" ),
974                       element );
975        }
976      }
977      else if ( ElementKind.METHOD == element.getKind() )
978      {
979        if ( !AnnotationsUtil.hasAnnotationOfType( element.getEnclosingElement(), Constants.FRAGMENT_CLASSNAME ) &&
980             !AnnotationsUtil.hasAnnotationOfType( element.getEnclosingElement(), Constants.INJECTOR_CLASSNAME ) )
981        {
982          reportError( env,
983                       MemberChecks.mustNot( Constants.NAMED_CLASSNAME,
984                                             "be a method unless the method is enclosed in a type annotated with " +
985                                             MemberChecks.toSimpleName( Constants.FRAGMENT_CLASSNAME ) + " or " +
986                                             MemberChecks.toSimpleName( Constants.INJECTOR_CLASSNAME ) ),
987                       element );
988        }
989      }
990      else
991      {
992        reportError( env,
993                     MemberChecks.toSimpleName( Constants.NAMED_CLASSNAME ) + " target is not valid",
994                     element );
995      }
996    }
997  }

1089  private void processFragment( @Nonnull final TypeElement element )
1090  {
1091    debug( () -> "Processing Fragment: " + element );
1092    if ( ElementKind.INTERFACE != element.getKind() )
1093    {
1094      throw new ProcessorException( MemberChecks.must( Constants.FRAGMENT_CLASSNAME, "be an interface" ),
1095                                    element );
1096    }
1097    else if ( !element.getTypeParameters().isEmpty() )
1098    {
1099      throw new ProcessorException( MemberChecks.mustNot( Constants.FRAGMENT_CLASSNAME, "have type parameters" ),
1100                                    element );
1101    }
1102    else if ( !element.getInterfaces().isEmpty() )
1103    {
1104      throw new ProcessorException( MemberChecks.mustNot( Constants.FRAGMENT_CLASSNAME, "extend any interfaces" ),
1105                                    element );
1106    }
1107    final List<IncludeDescriptor> includes = extractIncludes( element, Constants.FRAGMENT_CLASSNAME );
1108    final Map<ExecutableElement, Binding> bindings = new LinkedHashMap<>();
1109    final List<ExecutableElement> methods =
1110      ElementsUtil.getMethods( element, processingEnv.getElementUtils(), processingEnv.getTypeUtils() );
1111    for ( final ExecutableElement method : methods )
1112    {
1113      processProvidesMethod( element, bindings, method );
1114    }
1115    if ( bindings.isEmpty() && includes.isEmpty() )
1116    {
1117      throw new ProcessorException( MemberChecks.must( Constants.FRAGMENT_CLASSNAME,
1118                                                       "contain one or more methods or one or more includes" ),
1119                                    element );
1120    }
1121    final List<? extends AnnotationMirror> scopedAnnotations = getScopedAnnotations( element );
1122    if ( !scopedAnnotations.isEmpty() )
1123    {
1124      throw new ProcessorException( MemberChecks.mustNot( Constants.FRAGMENT_CLASSNAME,
1125                                                          "be annotated with an annotation that is " +
1126                                                          "annotated with the " + Constants.JSR_330_SCOPE_CLASSNAME +
1127                                                          " annotation such as " + scopedAnnotations ),
1128                                    element );
1129    }
1130    _registry.registerFragment( new FragmentDescriptor( element, includes, bindings.values() ) );
1131  }

1133  @SuppressWarnings( "unchecked" )
1134  @Nonnull
1135  private List<InputDescriptor> extractInputs( @Nonnull final TypeElement element )
1136  {
1137    final List<InputDescriptor> results = new ArrayList<>();
1138    final AnnotationMirror annotation = AnnotationsUtil.getAnnotationByType( element, Constants.INJECTOR_CLASSNAME );
1139    final AnnotationValue inputsAnnotationValue = AnnotationsUtil.findAnnotationValue( annotation, "inputs" );
1140    assert null != inputsAnnotationValue;
1141    final List<AnnotationMirror> inputs = (List<AnnotationMirror>) inputsAnnotationValue.getValue();
1142
1143    final int size = inputs.size();
1144    for ( int i = 0; i < size; i++ )
1145    {
1146      final AnnotationMirror input = inputs.get( i );
1147      final String qualifier = AnnotationsUtil.getAnnotationValueValue( input, "qualifier" );
1148      final AnnotationValue typeAnnotationValue = AnnotationsUtil.getAnnotationValue( input, "type" );
1149      final TypeMirror type = (TypeMirror) typeAnnotationValue.getValue();
1150      if ( TypeKind.ARRAY == type.getKind() )
1151      {
1152        throw new ProcessorException( MemberChecks.toSimpleName( Constants.INPUT_CLASSNAME ) +
1153                                      " must not specify an array type for the type parameter",
1154                                      element,
1155                                      input,
1156                                      typeAnnotationValue );
1157      }
1158      else if ( TypeKind.VOID == type.getKind() )
1159      {
1160        throw new ProcessorException( MemberChecks.toSimpleName( Constants.INPUT_CLASSNAME ) +
1161                                      " must specify a non-void type for the type parameter",
1162                                      element,
1163                                      input,
1164                                      typeAnnotationValue );
1165      }
1166      else if ( TypeKind.DECLARED == type.getKind() &&
1167                !( (TypeElement) ( (DeclaredType) type ).asElement() ).getTypeParameters().isEmpty() )
1168      {
1169        throw new ProcessorException( MemberChecks.toSimpleName( Constants.INPUT_CLASSNAME ) +
1170                                      " must not specify a parameterized type for the type parameter",
1171                                      element,
1172                                      input,
1173                                      typeAnnotationValue );
1174      }
1175      final Coordinate coordinate = new Coordinate( qualifier, type );
1176      final boolean optional = AnnotationsUtil.getAnnotationValueValue( input, "optional" );
1177      final ServiceSpec service = new ServiceSpec( coordinate, optional );
1178      final Binding binding =
1179        new Binding( Binding.Kind.INPUT,
1180                     element.getQualifiedName() + "#" + i,
1181                     Collections.singletonList( service ),
1182                     true,
1183                     element,
1184                     new ServiceDescriptor[ 0 ] );
1185      results.add( new InputDescriptor( service, binding, "input" + ( i + 1 ) ) );
1186    }
1187    return results;
1188  }

1190  @Nonnull
1191  private List<IncludeDescriptor> extractIncludes( @Nonnull final TypeElement element,
1192                                                   @Nonnull final String annotationClassname )
1193  {
1194    final List<IncludeDescriptor> results = new ArrayList<>();
1195    final List<TypeMirror> includes =
1196      AnnotationsUtil.getTypeMirrorsAnnotationParameter( element, annotationClassname, "includes" );
1197    for ( final TypeMirror include : includes )
1198    {
1199      final Element includeElement = processingEnv.getTypeUtils().asElement( include );
1200      if ( AnnotationsUtil.hasAnnotationOfType( includeElement, Constants.FRAGMENT_CLASSNAME ) ||
1201           AnnotationsUtil.hasAnnotationOfType( includeElement, Constants.INJECTABLE_CLASSNAME ) )
1202      {
1203        results.add( new IncludeDescriptor( (DeclaredType) include, include.toString() ) );
1204      }
1205      else
1206      {
1207        final ElementKind kind = includeElement.getKind();
1208        if ( ElementKind.CLASS == kind || ElementKind.INTERFACE == kind )
1209        {
1210          final List<ProviderEntry> providers =
1211            includeElement.getAnnotationMirrors()
1212              .stream()
1213              .map( a -> {
1214                final AnnotationMirror provider =
1215                  getStingProvider( element, annotationClassname, (TypeElement) includeElement, a );
1216                return null != provider ? new ProviderEntry( a, provider ) : null;
1217              } )
1218              .filter( Objects::nonNull )
1219              .collect( Collectors.toList() );
1220          if ( providers.size() > 1 )
1221          {
1222            final String message =
1223              MemberChecks.toSimpleName( annotationClassname ) + " target has an " +
1224              "'includes' parameter containing the value " + includeElement.asType() +
1225              " that is annotated by multiple @StingProvider annotations. Matching annotations:\n" +
1226              providers
1227                .stream()
1228                .map( a -> ( (TypeElement) a.getAnnotation().getAnnotationType().asElement() ).getQualifiedName() )
1229                .map( a -> "  " + a )
1230                .collect( Collectors.joining( "\n" ) );
1231            throw new ProcessorException( message, element );
1232          }
1233          else if ( !providers.isEmpty() )
1234          {
1235            final ProviderEntry entry = providers.get( 0 );
1236            final AnnotationMirror providerAnnotation = entry.getProvider();
1237            final String namePattern = AnnotationsUtil.getAnnotationValueValue( providerAnnotation, "value" );
1238
1239            final String targetCompoundType =
1240              namePattern
1241                .replace( "[SimpleName]", includeElement.getSimpleName().toString() )
1242                .replace( "[CompoundName]", getComponentName( (TypeElement) includeElement ) )
1243                .replace( "[EnclosingName]", getEnclosingName( (TypeElement) includeElement ) )
1244                .replace( "[FlatEnclosingName]", getEnclosingName( (TypeElement) includeElement ).replace( '.', '_' ) );
1245
1246            final String targetQualifiedName =
1247              ElementsUtil.getPackageElement( includeElement ).getQualifiedName().toString() + "." + targetCompoundType;
1248
1249            results.add( new IncludeDescriptor( (DeclaredType) include, targetQualifiedName ) );
1250          }
1251          else
1252          {
1253            throw new ProcessorException( MemberChecks.toSimpleName( annotationClassname ) +
1254                                          " target has an includes parameter containing the value " + include +
1255                                          " that is not a type annotated by either " +
1256                                          MemberChecks.toSimpleName( Constants.FRAGMENT_CLASSNAME ) +
1257                                          " or " +
1258                                          MemberChecks.toSimpleName( Constants.INJECTABLE_CLASSNAME ) +
1259                                          " and the type does not declare a provider",
1260                                          element );
1261          }
1262        }
1263      }
1264    }
1265    return results;

1353  private void processProvidesMethod( @Nonnull final TypeElement element,
1354                                      @Nonnull final Map<ExecutableElement, Binding> bindings,
1355                                      @Nonnull final ExecutableElement method )
1356  {
1357    if ( TypeKind.VOID == method.getReturnType().getKind() )
1358    {
1359      throw new ProcessorException( MemberChecks.must( Constants.FRAGMENT_CLASSNAME,
1360                                                       "only contain methods that return a value" ),
1361                                    method );
1362    }
1363    if ( !method.getTypeParameters().isEmpty() )
1364    {
1365      throw new ProcessorException( MemberChecks.mustNot( Constants.FRAGMENT_CLASSNAME,
1366                                                          "contain methods with a type parameter" ),
1367                                    method );
1368    }
1369    if ( !method.getModifiers().contains( Modifier.DEFAULT ) )
1370    {
1371      throw new ProcessorException( MemberChecks.must( Constants.FRAGMENT_CLASSNAME,
1372                                                       "only contain methods with a default modifier" ),
1373                                    method );
1374    }
1375    final boolean nullablePresent = AnnotationsUtil.hasNullableAnnotation( method );
1376    if ( nullablePresent && method.getReturnType().getKind().isPrimitive() )
1377    {
1378      throw new ProcessorException( MemberChecks.toSimpleName( Constants.FRAGMENT_CLASSNAME ) +
1379                                    " contains a method that is incorrectly annotated with " +
1380                                    MemberChecks.toSimpleName( AnnotationsUtil.NULLABLE_CLASSNAME ) +
1381                                    " as the return type is a primitive value",
1382                                    method );
1383    }
1384    final List<? extends AnnotationMirror> scopedAnnotations = getScopedAnnotations( method );
1385    if ( !scopedAnnotations.isEmpty() )
1386    {
1387      throw new ProcessorException( MemberChecks.mustNot( Constants.FRAGMENT_CLASSNAME,
1388                                                          "contain a method that is annotated with an " +
1389                                                          "annotation that is annotated with the " +
1390                                                          Constants.JSR_330_SCOPE_CLASSNAME +
1391                                                          " annotation such as " + scopedAnnotations ),
1392                                    method );
1393    }
1394
1395    final boolean eager = AnnotationsUtil.hasAnnotationOfType( method, Constants.EAGER_CLASSNAME );
1396
1397    final List<ServiceDescriptor> dependencies = new ArrayList<>();
1398    int index = 0;
1399    final List<? extends TypeMirror> parameterTypes = ( (ExecutableType) method.asType() ).getParameterTypes();
1400    for ( final VariableElement parameter : method.getParameters() )
1401    {
1402      dependencies.add( processFragmentServiceParameter( parameter, parameterTypes.get( index ), index ) );
1403      index++;
1404    }
1405
1406    final AnnotationMirror annotation = AnnotationsUtil.findAnnotationByType( method, Constants.TYPED_CLASSNAME );
1407    final AnnotationValue value =
1408      null != annotation ? AnnotationsUtil.findAnnotationValue( annotation, "value" ) : null;
1409
1410    final String qualifier = getQualifier( method );
1411    if ( AnnotationsUtil.hasAnnotationOfType( method, Constants.JSR_330_NAMED_CLASSNAME ) )
1412    {
1413      final String message =
1414        MemberChecks.mustNot( Constants.FRAGMENT_CLASSNAME,
1415                              "contain a method annotated with the " + Constants.JSR_330_NAMED_CLASSNAME +
1416                              " annotation. Use the " + Constants.NAMED_CLASSNAME + " annotation instead" );
1417      throw new ProcessorException( message, method );
1418    }
1419    if ( AnnotationsUtil.hasAnnotationOfType( method, Constants.CDI_TYPED_CLASSNAME ) )
1420    {
1421      final String message =
1422        MemberChecks.mustNot( Constants.FRAGMENT_CLASSNAME,
1423                              "contain a method annotated with the " + Constants.CDI_TYPED_CLASSNAME +
1424                              " annotation. Use the " + Constants.TYPED_CLASSNAME + " annotation instead" );
1425      throw new ProcessorException( message, method );
1426    }
1427
1428    @SuppressWarnings( "unchecked" )
1429    final List<TypeMirror> types =
1430      null == value ?
1431      Collections.singletonList( method.getReturnType() ) :
1432      ( (List<AnnotationValue>) value.getValue() )
1433        .stream()
1434        .map( v -> (TypeMirror) v.getValue() )
1435        .collect( Collectors.toList() );
1436
1437    final ServiceSpec[] specs = new ServiceSpec[ types.size() ];
1438    for ( int i = 0; i < specs.length; i++ )
1439    {
1440      final TypeMirror type = types.get( i );
1441      if ( !processingEnv.getTypeUtils().isAssignable( method.getReturnType(), type ) )
1442      {
1443        throw new ProcessorException( MemberChecks.toSimpleName( Constants.TYPED_CLASSNAME ) +
1444                                      " specified a type that is not assignable to the return type of the method",
1445                                      element,
1446                                      annotation,
1447                                      value );
1448      }
1449      else if ( TypeKind.DECLARED == type.getKind() && isParameterized( (DeclaredType) type ) )
1450      {
1451        throw new ProcessorException( MemberChecks.toSimpleName( Constants.TYPED_CLASSNAME ) +
1452                                      " specified a type that is a a parameterized type",
1453                                      element,
1454                                      annotation,
1455                                      value );
1456      }
1457      specs[ i ] = new ServiceSpec( new Coordinate( qualifier, type ), nullablePresent );
1458    }
1459
1460    if ( 0 == specs.length && !eager )
1461    {
1462      throw new ProcessorException( MemberChecks.mustNot( Constants.FRAGMENT_CLASSNAME,
1463                                                          "contain methods that specify zero types with the " +
1464                                                          MemberChecks.toSimpleName( Constants.TYPED_CLASSNAME ) +
1465                                                          " annotation and are not annotated with the " +
1466                                                          MemberChecks.toSimpleName( Constants.EAGER_CLASSNAME ) +
1467                                                          " annotation otherwise the component can not be created by " +
1468                                                          "the injector" ),
1469                                    element );
1470    }
1471    if ( 0 == specs.length && !"".equals( qualifier ) )
1472    {
1473      throw new ProcessorException( MemberChecks.mustNot( Constants.FRAGMENT_CLASSNAME,
1474                                                          "contain methods that specify zero types with the " +
1475                                                          MemberChecks.toSimpleName( Constants.TYPED_CLASSNAME ) +
1476                                                          " annotation and specify a qualifier with the " +
1477                                                          MemberChecks.toSimpleName( Constants.NAMED_CLASSNAME ) +
1478                                                          " annotation as the qualifier is meaningless" ),
1479                                    element );
1480    }
1481
1482    final Binding binding =
1483      new Binding( Binding.Kind.PROVIDES,
1484                   element.getQualifiedName() + "#" + method.getSimpleName(),
1485                   Arrays.asList( specs ),
1486                   eager,
1487                   method,
1488                   dependencies.toArray( new ServiceDescriptor[ 0 ] ) );
1489    bindings.put( method, binding );

1616         ElementsUtil.isWarningNotSuppressed( element, Constants.WARNING_JSR_330_NAMED ) )
1617    {
1618      final String message =
1619        MemberChecks.mustNot( Constants.INJECTABLE_CLASSNAME,
1620                              "be annotated with the " + Constants.JSR_330_NAMED_CLASSNAME + " annotation. " +
1621                              "Use the " + Constants.NAMED_CLASSNAME + " annotation instead. " +
1622                              MemberChecks.suppressedBy( Constants.WARNING_JSR_330_NAMED ) );

1631                              "Use the " + Constants.TYPED_CLASSNAME + " annotation instead. " +
1632                              MemberChecks.suppressedBy( Constants.WARNING_CDI_TYPED ) );

1634    }

1636         ElementsUtil.isWarningNotSuppressed( constructor, Constants.WARNING_JSR_330_INJECT ) )

1638      final String message =
1639        MemberChecks.mustNot( Constants.INJECTABLE_CLASSNAME,

1641                              MemberChecks.suppressedBy( Constants.WARNING_JSR_330_INJECT ) );
1642      processingEnv.getMessager().printMessage( Diagnostic.Kind.WARNING, message, constructor );
1643    }
1644    @SuppressWarnings( "unchecked" )
1645    final List<TypeMirror> types =
1646      null == value ?
1647      Collections.singletonList( element.asType() ) :
1648      ( (List<AnnotationValue>) value.getValue() )
1649        .stream()
1650        .map( v -> (TypeMirror) v.getValue() )
1651        .collect( Collectors.toList() );
1652
1653    final ServiceSpec[] specs = new ServiceSpec[ types.size() ];
1654    for ( int i = 0; i < specs.length; i++ )
1655    {
1656      final TypeMirror type = types.get( i );

1699                   element.getQualifiedName().toString(),
1700                   Arrays.asList( specs ),
1701                   eager,
1702                   constructor,
1703                   dependencies.toArray( new ServiceDescriptor[ 0 ] ) );

1844            MemberChecks.mustNot( Constants.INJECTABLE_CLASSNAME,
1845                                  "contain a constructor with a parameter annotated with the " +
1846                                  Constants.JSR_330_NAMED_CLASSNAME + " annotation. Use the " +
1847                                  Constants.NAMED_CLASSNAME + " annotation instead. " +
1848                                  MemberChecks.suppressedBy( Constants.WARNING_JSR_330_NAMED ) );
1849          processingEnv.getMessager().printMessage( Diagnostic.Kind.WARNING, message, parameter );
1850        }
1851        final Coordinate coordinate = new Coordinate( qualifier, dependencyType );
1852        final ServiceSpec service = new ServiceSpec( coordinate, optional );
1853        return new ServiceDescriptor( kind, service, parameter, parameterIndex );
1854      }
1855    }
1856  }

1925  @Nullable
1926  private byte[] tryLoadDescriptorData( @Nonnull final TypeElement element )
1927  {
1928    byte[] data = tryLoadDescriptorData( StandardLocation.CLASS_PATH, element );
1929    data = null != data ? data : tryLoadDescriptorData( StandardLocation.CLASS_OUTPUT, element );
1930    // Some tools (IDEA?) will actually put dependencies on the boot/platform class path. This
1931    // seems like it should be an error but as long as the tools do this, the annotation processor
1932    // must also be capable of loading descriptor data from the platform classpath
1933    return null != data ? data : tryLoadDescriptorData( StandardLocation.PLATFORM_CLASS_PATH, element );
1934  }

1936  @Nullable
1937  private byte[] tryLoadDescriptorData( @Nonnull final JavaFileManager.Location location,
1938                                        @Nonnull final TypeElement element )
1939  {
1940    final String[] nameParts = extractNameParts( element );
1941    try
1942    {
1943      return IOUtil.readFully( processingEnv.getFiler().getResource( location, nameParts[ 0 ], nameParts[ 1 ] ) );
1944    }
1945    catch ( final IOException ignored )
1946    {
1947      return null;
1948    }
1949    catch ( final RuntimeException e )
1950    {
1951      // The javac compiler in Java8 will return a null from the JavaFileManager when it should
1952      // throw an IOException which later causes a NullPointerException when wrapping the code
1953      // This ugly hack works around this scenario and just lets the compile continue
1954      if ( e.getClass().getCanonicalName().equals( "com.sun.tools.javac.util.ClientCodeException" ) &&
1955           e.getCause() instanceof NullPointerException )
1956      {
1957        return null;
1958      }
1959      else
1960      {
1961        throw e;
1962      }
1963    }
1964  }
